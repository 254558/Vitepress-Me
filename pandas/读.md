# pandas

``` py
import pandas as pd

# 1. 创建数据
data = {
    "姓名": ["小明", "小红", "小刚", "小李"],
    "语文": [85, 92, 78, 90],
    "数学": [90, 88, 95, 82]
}
df = pd.DataFrame(data)

# 2. 新增总分
df["总分"] = df["语文"] + df["数学"]

# 3. 筛选总分 > 170 的学生
result = df[df["总分"] > 170]

# 4. 按总分排序
result = result.sort_values(by="总分", ascending=False)

# 5. 输出结果
print(result)
```
